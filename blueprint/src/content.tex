\chapter*{Introduction}

This blueprint accompanies a Rocq formalisation that machine-verifies the
numerical step underpinning Maynard's bounded-gap theorem:
\[
   \Mk[105] \;=\; 105 \cdot \lammax\!\bigl(\Mone^{-1}\Mtwo\bigr) \;>\; 4.
\]
The headline is formula~(8.15) of James Maynard,
\emph{Small gaps between primes} (arXiv:1311.4600;
Annals of Mathematics \textbf{181} (2015), 383--413).
In the published proof, this inequality is checked by a Mathematica
notebook (\texttt{Computations.nb}, supplied as supplementary material
with the arXiv preprint). The present project replaces that
Mathematica-based computation with a Rocq proof that re-derives every
algebraic claim inside the kernel.

The end-to-end statement is
\begin{verbatim}
Theorem maynard_M105_certified :
  MaynardVerify.all_match_M1Z = true /\
  MaynardVerify.all_match_M2Z = true /\
  exists lambda : realalg,
    eigenvalue (map_mx (ratr : rat -> realalg) A_rat) lambda
    /\ (ratr (4%:Q / 105%:Q) : realalg) < lambda.
\end{verbatim}
proven in \texttt{theories/S1/Cert.v}. \texttt{Print Assumptions} on this
theorem reports only Rocq's \texttt{PrimInt63} kernel primitives:
the project itself contains zero \texttt{Axiom} declarations and zero
\texttt{Admitted} lemmas.

\paragraph{Scope.}
What this Rocq layer proves, kernel-checked, is the left-hand side of
\[
\underbrace{\;\lammax(\Mone^{-1}\Mtwo) \;>\; 4/105\;}_{\text{Rocq, kernel-checked}}
\quad\Longrightarrow\quad
\underbrace{\;\Mk[105] = 105\cdot\lammax \;>\; 4\;}_{\text{Maynard's paper, Lemma 8.3}}.
\]
The right arrow is Maynard's Lemma~8.3 (a generalised Rayleigh-quotient
identity proved in the Annals paper, refereed there); the project does
not re-formalise this analytic step. The project's scope is exactly to
replace the Mathematica computation, not to re-formalise Maynard's paper.

\paragraph{Two distinct certificates for the same fact.}
Maynard's notebook proves $\lammax > 4/k$ via the \emph{eigenvector
route}: it numerically computes the top eigenvector of $\Mone^{-1}\Mtwo$,
snaps it to a small-denominator rational vector, and evaluates the
Rayleigh quotient $k\cdot v^T M_2 v / v^T M_1 v$ in exact rational
arithmetic. This is a true rigorous lower bound regardless of how~$v$
is obtained. Our Rocq proof takes the \emph{characteristic-polynomial
route}: a Brown--Traub Sturm chain on $\charpolysym_{\Arat}$ certifies
the existence of an eigenvalue strictly above $4/105$ via 1-D root
counting (intermediate value theorem). No eigenvector is constructed.
Both routes are sound; together, they form an independent
double-check of Maynard's headline value $\Mk[105] \approx 4.0021$.

\chapter{The matrix construction}\label{ch:matrix}

This chapter sets up the input data that the rest of the proof consumes:
the 42-element basis, the closed-form entries of $\Mone$ and $\Mtwo$,
the $\Z$-level cross-multiplication checks that anchor the
FLINT-shipped integer matrices to those closed forms, the kernel-Qed
bridge between the rat-level paper-form spec and the Z-level
computational spec, and the bridge into MathComp's $\Q$-level matrices.
All material here lives inside the files \texttt{MaynardSpec.v},
\texttt{MaynardBasis.v}, \texttt{MaynardFactQ.v}, the
\texttt{MaynardVerify/} directory (\texttt{Def.v}, six parallel
\texttt{M2\_0..5.v} chunks, and the assembly \texttt{MaynardVerify.v}),
and \texttt{MaynardSpecBridge.v}. Throughout this chapter we follow the
v3/Annals numbering of Maynard's paper: Lemma~8.1 / 8.2 / 8.3 (=
Lemma~7.1 / 7.2 / 7.3 in v1).

\section{Conventions}

\begin{convention}
The two power sums are
$P_1 = \sum_i t_i$ and $P_2 = \sum_i t_i^2$. The basis monomials are
$F_{b,c} = (1 - P_1)^b \cdot P_2^c$ with $b + 2c \le 11$. Indices
$(b_i, c_i)$, $(b_j, c_j)$ denote bidegrees of the $i$-th and $j$-th
basis monomials. The outer integration dimension is $k = 105$; we use
two constants $K_1 = 105$ (outer simplex) and $K_2 = 104 = K_1 - 1$
(residual simplex after the eq.~8.8 substitution that integrates out
$t_1$).
\end{convention}

\begin{definition}[Constants $K_1$, $K_2$]\label{def:K1K2}
\rocq{PrimeGapS1.MaynardSpec.K1, PrimeGapS1.MaynardSpec.K2}
\rocqok
$K_1 := 105$, $K_2 := 104$. The first is the outer simplex dimension
(used in $M_1$ and in the threshold $4/K_1$); the second is the
residual simplex dimension after the eq.~8.8 substitution (used in $M_2$).
\end{definition}

\section{The 42-element basis}

\begin{definition}[Basis]\label{def:maynard_basis}
\rocq{PrimeGapS1.MaynardBasis.maynard_basis}
\rocqok
The Maynard basis is the explicit list of 42 pairs $(b, c) \in \N^2$
with $b + 2c \le 11$, in the Mathematica enumeration order used by
\texttt{flint\_probe.py}:
\[
[(0,0), (1,0), (0,1), (2,0), (1,1), (3,0),\, (0,2), \ldots, (9,1), (11,0)].
\]
\end{definition}

\begin{lemma}[Basis matches FLINT enumeration]\label{lem:basis_matches}
\rocq{PrimeGapS1.MaynardBasis.maynard_basis_eq_witness}
\rocqok
\uses{def:maynard_basis}
$\texttt{maynard\_basis} = \texttt{Witness.basis}$,
i.e.\ the hand-readable basis in \texttt{MaynardBasis.v} agrees
literally with the autogenerated basis used to index the shipped
integer matrices in \texttt{Witness.v}.
\end{lemma}
\begin{proof}\rocqok
\vmcompute{} on a list of 42 pairs.
\end{proof}

\begin{remark}
Order matters only because the matrix rows and columns must align with
the indexing of $M_{1,2}^{\mathrm{int}}$; eigenvalues of $M_1^{-1} M_2$
are invariant under simultaneous row/column permutation, so any
ordering of the same set yields the same $\lammax$.
\end{remark}

\begin{remark}
This is a lower-bound subspace argument. Maynard's full optimisation is
over all symmetric $F$ of degree $\le 11$ in $k = 105$ variables;
restricting to polynomials in the two power sums $P_1, P_2$ is a
strict subspace, and the supremum of a Rayleigh quotient over a
subspace is $\le$ the unrestricted supremum. So a lower bound of
$4/105$ on the 42-dimensional restricted problem yields
$\Mk[105] > 4$.
\end{remark}

\section{The polynomial \texorpdfstring{$G_{n,2}(k)$}{G\_n,2(k)} (Lemma 8.1)}

\begin{definition}[Compositions of $n$]\label{def:compositions}
\rocq{PrimeGapS1.MaynardSpec.compositions}
\rocqok
For $r, n \in \N$,
$\Comp(r, n) := \{(b_1, \ldots, b_r) \in \N_{\ge 1}^r : \sum_s b_s = n\}$,
returned as a list in lexicographic order.  For $r = 0$ this is the
singleton $\{[\,]\}$ if $n = 0$ and empty otherwise; for $r = n + 1$
or larger it is empty.
\end{definition}

\begin{definition}[Inner factor $\Cff(a)$]\label{def:cff}
\rocq{PrimeGapS1.MaynardSpec.cff}
\rocqok
\uses{def:compositions}
For a composition $a = (b_1, \ldots, b_r)$,
\[
\Cff(a) \;:=\; \prod_{s=1}^{r} \frac{(2 b_s)!}{b_s!}.
\]
This is exactly the per-composition inner product appearing in
Maynard's Lemma~8.1 (with $j = 2$, so $j b_s = 2 b_s$).
\end{definition}

\begin{definition}[Maynard's $G_{n,2}(k)$]\label{def:G_2}
\rocq{PrimeGapS1.MaynardSpec.G_2}
\rocqok
\uses{def:compositions, def:cff}
For $n, k \in \N$:
$\Gnk{0}{k} = 1$, and for $n \ge 1$,
\[
\Gnk{n}{k} \;=\; n! \cdot \sum_{r = 1}^{n} \binom{k}{r}
                          \sum_{a \in \Comp(r, n)} \Cff(a).
\]
This is Lemma~8.1 of arXiv:1311.4600~v3 character-for-character: the
$n!$ prefactor, the outer $r$-sum, the binomial coefficient, the
inner sum over length-$r$ compositions of $n$, and the inner product
$\prod_s (j b_s)!/b_s!$ with $j = 2$ all appear at the same positions.
\end{definition}

\begin{lemma}[Sanity values of $G_{n,2}$]\label{lem:G_2_small}
\rocqok
\uses{def:G_2}
$\Gnk{0}{k} = 1$, $\Gnk{1}{k} = 2k$, $\Gnk{2}{k} = 4 k^2 + 20 k$.
\end{lemma}
\begin{proof}\rocqok
$\Gnk{1}{k}$: the only length-1 composition of $1$ is $[1]$, with
$\Cff([1]) = 2!/1! = 2$, so the sum is $1! \cdot \binom{k}{1} \cdot 2 = 2k$.
$\Gnk{2}{k}$: length-1 contributes $\binom{k}{1} \cdot \Cff([2]) = k \cdot 4!/2! = 12k$;
length-2 contributes $\binom{k}{2} \cdot \Cff([1,1]) = \binom{k}{2} \cdot (2 \cdot 2) = 2k(k-1)$;
total $= 2! \cdot (12k + 2k(k-1)) = 2(12k + 2k^2 - 2k) = 4k^2 + 20k$.
Both also reduce by \vmcompute{}.
\end{proof}

\section{Closed forms for \texorpdfstring{$M_1$ and $M_2$}{M\_1 and M\_2} (Lemma 8.2)}

\begin{definition}[$M_1$ entry]\label{def:M1_entry}
\rocq{PrimeGapS1.MaynardSpec.M1_entry}
\rocqok
\uses{def:G_2, def:K1K2}
For $(b_i, c_i, b_j, c_j)$ basis bidegrees, set $b = b_i + b_j$ and
$c = c_i + c_j$. Then
\[
M_{1, ij} \;=\; \frac{b!}{(K_1 + b + 2c)!} \cdot \Gnk{c}{K_1}.
\]
This is Lemma~8.2 (first part) of arXiv:1311.4600~v3 verbatim, with
$K_1 = k = 105$.
\end{definition}

\begin{definition}[Eq.~8.8 expansion coefficient $\alpha$]\label{def:alpha}
\rocq{PrimeGapS1.MaynardSpec.alpha}
\rocqok
For $b, c, c' \in \N$ with $c' \le c$,
\[
\alpha(b, c, c') \;=\;
   \frac{\binom{c}{c'} \cdot b! \cdot (2c - 2c')!}{(b + 2c - 2c' + 1)!}.
\]
This is the coefficient of
$(1 - \sum_{j \ne 1} t_j)^{b + 2c - 2c' + 1} \cdot (\sum_{j \ne 1} t_j^2)^{c'}$
in the expansion of
$\int_0^{1 - \sum_{j \ne 1} t_j} (1 - P_1)^b P_2^c \, dt_1$,
obtained by expanding $P_2^c = (t_1^2 + \sum_{j \ne 1} t_j^2)^c$
binomially and integrating $t_1^{2(c - c')} (1 - \sum_j t_j)^b$
via the Beta-function identity
$\int_0^X t^a (X - t)^b \, dt = a! b! X^{a + b + 1} / (a + b + 1)!$.
\end{definition}

\begin{definition}[$M_2$ entry]\label{def:M2_entry}
\rocq{PrimeGapS1.MaynardSpec.M2_entry}
\rocqok
\uses{def:alpha, def:G_2, def:K1K2}
For $(b_i, c_i, b_j, c_j)$ basis bidegrees, with
$b'_1 = b_i + 2c_i - 2c'_1 + 1$, $b'_2 = b_j + 2c_j - 2c'_2 + 1$,
$b_{\mathrm{sum}} = b'_1 + b'_2$, $c_{\mathrm{sum}} = c'_1 + c'_2$:
\[
M_{2, ij}^{(1)} \;=\;
   \sum_{c'_1 = 0}^{c_i} \sum_{c'_2 = 0}^{c_j}
     \alpha(b_i, c_i, c'_1)\cdot \alpha(b_j, c_j, c'_2)
     \cdot \frac{b_{\mathrm{sum}}!}{(K_2 + b_{\mathrm{sum}} + 2 c_{\mathrm{sum}})!}
     \cdot \Gnk{c_{\mathrm{sum}}}{K_2}.
\]
This is Lemma~8.2 (second part) of arXiv:1311.4600~v3 for the
\emph{per-coordinate} integrand $J_k^{(1)}$ (not $\sum_m J_k^{(m)}$):
the symmetry factor $k$ is paid in the threshold $4/k$, not in the
matrix entries.
\end{definition}

\begin{remark}
The matrices have up to $36$ rational summands per entry (when
$c_i = c_j = 5$), with intermediate denominators reaching
$\sim 10^{7000}$ digits before the rational normalisation collapses
them. This is why \texttt{MaynardSpec.v} also ships
$\Z$-level twins (\texttt{m1\_num\_den}, \texttt{m2\_num\_den}) that
keep the $(\text{num}, \text{den})$ pair separate; \texttt{MaynardVerify.v}
operates on those twins to avoid \texttt{rat}'s
canonical \texttt{gcd}-normalisation cost during \vmcompute.
\end{remark}

\section{Kernel cross-check: matrices match the spec}

\begin{theorem}[$M_1$ entries match Maynard's closed form]\label{thm:M1_match}
\rocq{PrimeGapS1.MaynardVerify.Def.all_match_M1Z_true}
\rocqok
\uses{def:M1_entry}
For every $(i, j) \in \{0, \ldots, 41\}^2$,
\[
\texttt{m1\_num}(i, j) \cdot \DMone
   \;=\; M_1^{\mathrm{int}}[i][j] \cdot \texttt{m1\_den}(i, j)\quad\text{in }\Z,
\]
where $M_1^{\mathrm{int}}$ is the FLINT-shipped integer matrix and
$(\texttt{m1\_num}, \texttt{m1\_den})$ are the closed-form
$(\text{numerator}, \text{denominator})$ from
Definition~\ref{def:M1_entry}.
\end{theorem}
\begin{proof}\rocqok
The $1764$ Boolean cross-multiplication checks
\texttt{M1\_entry\_matchZ}~$i\,j$ are aggregated into
\texttt{all\_match\_M1Z}~$=$~\texttt{forallb}~over $42 \times 42$, and
the Lemma is closed by a single \vmcompute{}. The check costs
${\sim}90$~s wall-clock. \texttt{Print Assumptions} reports only the
standard \Uintsf{} primitives.
\end{proof}

\begin{theorem}[$M_2$ entries match Maynard's closed form]\label{thm:M2_match}
\rocq{PrimeGapS1.MaynardVerify.all_match_M2Z_true}
\rocqok
\uses{def:M2_entry}
The analogous identity holds for $M_2$:
$\texttt{m2\_num}(i, j) \cdot \DMtwo
   = M_2^{\mathrm{int}}[i][j] \cdot \texttt{m2\_den}(i, j)$
for all $(i, j) \in \{0, \ldots, 41\}^2$.
\end{theorem}
\begin{proof}\rocqok
Same shape as Theorem~\ref{thm:M1_match}, but the entries are double
sums of up to 36 rational terms, so the per-row \vmcompute{} cost is
significant. The check is split into six 7-row chunks
(\texttt{MaynardVerify/M2\_0..5.v}) that \texttt{make -j} runs
concurrently, and the assembly stitches them via \texttt{seq\_split\_42}
plus \texttt{forallb\_app} — no per-entry recomputation in the
\Qed{} of \texttt{all\_match\_M2Z\_true} itself.
\texttt{Print Assumptions}: only the standard \Uintsf{} primitives.
\end{proof}

\begin{remark}
Theorems~\ref{thm:M1_match} and~\ref{thm:M2_match} together close the
trust gap on the input data: every entry of the $42 \times 42$ integer
matrices that the rest of the proof consumes is kernel-verified to
agree with Maynard's closed-form specification (Lemma~8.2). The FLINT
generator is therefore not part of the trust base for the input data
either.
\end{remark}

\section{Paper-form $\leftrightarrow$ computational-form spec bridge}

The cross-checks above operate on the $\Z$-level twins
(\texttt{m1\_num\_den}, \texttt{m2\_num\_den}) because \vmcompute{} on
$\rat$-level matrices stalls on canonical-structure elaboration
(see~\S\ref{ch:hb}). The rat-level spec
(\texttt{M1\_entry}, \texttt{M2\_entry}) is the
documentation-shaped form a reviewer reads side by side with
arXiv:1311.4600~\S8. The bridge below certifies in the kernel that the
two forms encode the same closed forms: a single \Qed{} per matrix,
with no axioms.

\begin{definition}[Reading a $\Z$-pair as a rational]\label{def:qfrac}
\rocq{PrimeGapS1.MaynardSpecBridge.qfrac}
\rocqok
For $(n, d) \in \Z \times \Z$, set
$\texttt{qfrac}(n, d) := \frac{n}{d} \in \rat$
(MathComp's int--to--rat morphism applied component-wise, then divided).
This reads a $(\text{numerator}, \text{denominator})$ pair as the
rational it represents.
\end{definition}

\begin{theorem}[$\Mone$ paper-form spec equals $\Z$-level spec]\label{thm:M1_spec_rat_eq}
\rocq{PrimeGapS1.MaynardSpecBridge.M1_spec_rat_eq}
\rocqok
\uses{def:M1_entry, def:qfrac}
For every $(i, j) \in \{0, \ldots, 41\}^2$,
\[
  \texttt{M1\_spec\_ij}(i, j) \;=\; \texttt{qfrac}\bigl(\texttt{m1\_num\_den\_at}(i, j)\bigr).
\]
\end{theorem}
\begin{proof}\rocqok
Layered structural induction in
\texttt{theories/S1/MaynardSpecBridge.v}: \texttt{factZ\_to\_rat},
\texttt{dblratZ\_to\_rat}, \texttt{binZ\_to\_rat},
\texttt{compositionsZ\_eq\_compositions}, \texttt{G2Z\_to\_rat}, and
\texttt{m1\_num\_den\_to\_rat} compose into a single rewrite chain.
\texttt{Print Assumptions} reports \emph{Closed under the global
context}: no axioms, not even \Uintsf\ primitives.
\end{proof}

\begin{theorem}[$\Mtwo$ paper-form spec equals $\Z$-level spec]\label{thm:M2_spec_rat_eq}
\rocq{PrimeGapS1.MaynardSpecBridge.M2_spec_rat_eq}
\rocqok
\uses{def:M2_entry, def:qfrac}
For every $(i, j) \in \{0, \ldots, 41\}^2$,
\[
  \texttt{M2\_spec\_ij}(i, j) \;=\; \texttt{qfrac}\bigl(\texttt{m2\_num\_den\_at}(i, j)\bigr).
\]
\end{theorem}
\begin{proof}\rocqok
Same layered induction, with two extra steps for the running
$\texttt{qplus}$ accumulator: \texttt{qfrac\_qmul} and
\texttt{qfrac\_qplus} bridge \texttt{qmul} / \texttt{qplus} on
$\Z \times \Z$ pairs to $*$ / $+$ on $\rat$, and a generic
$\texttt{fold\_left qplus} \to \sum$ lemma collects the double sum
over $(c'_1, c'_2)$. \texttt{Print Assumptions}: \emph{Closed under
the global context}.
\end{proof}

\begin{remark}
Theorems~\ref{thm:M1_spec_rat_eq} and~\ref{thm:M2_spec_rat_eq} live
in \texttt{MaynardSpecBridge.v}, a leaf in the dependency DAG:
\texttt{Cert.v} does not import them, and the headline theorem does
not depend on them. Their job is purely to certify that the
documentation-shaped $\rat$-level definitions
(Definitions~\ref{def:M1_entry} and~\ref{def:M2_entry}, written to
match Maynard's paper character for character) and the
$\vmcompute$-shaped $\Z$-level definitions
(\texttt{m1\_num\_den}, \texttt{m2\_num\_den}, consumed by
Theorems~\ref{thm:M1_match} and~\ref{thm:M2_match}) compute the same
rational. Combined with those theorems, this gives a two-step
kernel-verified path from each shipped integer matrix entry back to
Maynard's exact closed form.
\end{remark}

\chapter{The characteristic polynomial route}\label{ch:charpoly}

This chapter develops the in-kernel characteristic polynomial
machinery. We use a hand-rolled Faddeev--LeVerrier (FL) recurrence on
\texttt{list (list Z)} matrices, prove that it computes the same
polynomial as MathComp's \texttt{char\_poly} on the corresponding
$\Q$-matrix, and develop the $c \cdot M$ scaling formula needed to
relate the cleared-denominator polynomial to the rational
$\charpolysym_{\Arat}$. The material lives in
\texttt{IntPoly.v}, \texttt{IntMat.v}, \texttt{CharPoly.v} and
\texttt{CharPolyScale.v}.

\section{Integer polynomial and matrix libraries}

\begin{definition}[Integer polynomial type]\label{def:pol}
\rocq{PrimeGapS1.IntPoly.pol}
\rocqok
$\texttt{pol} := \texttt{list }\Z$, low-to-high coefficient order. The
operations \texttt{padd}, \texttt{psub}, \texttt{pneg}, \texttt{pscale},
\texttt{pmul}, \texttt{pderiv}, \texttt{prem} (pseudo-remainder),
\texttt{plead}, \texttt{pdeg}, \texttt{pnorm} (strip trailing zeros),
\texttt{peval} and \texttt{peval\_at\_rat} are all
\vmcompute-reducible and proved correct against ordinary integer
arithmetic.
\end{definition}

\begin{definition}[Integer matrix type]\label{def:mat}
\rocq{PrimeGapS1.IntMat.mat}
\rocqok
$\texttt{mat} := \texttt{list (list }\Z\texttt{)}$, row-major, no
well-formedness invariant. The operations \texttt{mzero},
\texttt{meye}, \texttt{madd}, \texttt{mscale}, \texttt{mmul},
\texttt{mtrans}, \texttt{mtrace}, \texttt{mat\_get} are pure list
recursions; nothing imports MathComp at this layer. The reason is that
MathComp's $'M[\Q]_{42}$ does not reduce under \vmcompute{} at the
sizes we need, but nested \texttt{list (list Z)} does
(${\sim}0.14$~s for a $42 \times 42$ multiplication).
\end{definition}

\section{Faddeev--LeVerrier on \texorpdfstring{$\Z$}{Z}-matrices}

\begin{definition}[FL recurrence]\label{def:fl_loop}
\rocq{PrimeGapS1.CharPoly.fl_loop, PrimeGapS1.CharPoly.char_poly_int}
\rocqok
\uses{def:mat, def:pol}
\texttt{fl\_loop} runs the classical Faddeev--LeVerrier recurrence
$M_0 := 0$, $c_0 := 1$, $M_k := A \cdot M_{k-1} + c_{k-1} I$,
$c_k := -\trace(A \cdot M_k)/k$, accumulating the coefficient list.
$\texttt{char\_poly\_int}(A) := \texttt{rev}(c_0, c_1, \ldots, c_n) \mathrel{++} [1]$
gives the monic char poly with leading coefficient~$1$.
\end{definition}

\begin{lemma}[FL agrees with $\Q$-FL]\label{lem:fl_loop_rat_is_charpoly}
\rocq{PrimeGapS1.CharPoly.fl_loop_rat_is_char_poly_L2,
      PrimeGapS1.CharPoly.fl_invariant_L2,
      PrimeGapS1.CharPoly.fl_divisibility_L2}
\rocqok
\uses{def:fl_loop}
After viewing $A : \texttt{mat}$ as the $\Q$-matrix
$\texttt{mat\_int\_to\_rat}(A, 1, n) : 'M[\Q]_n$, the integer FL
recurrence reproduces the rational FL recurrence step by step. The
auxiliary invariants \texttt{fl\_invariant\_L2} (the
relationship between $M_k$, $c_k$ and the rational $M_k^{\Q}$) and
\texttt{fl\_divisibility\_L2} (the classical Newton-identity fact that
$\trace(A \cdot M_k)$ is divisible by~$k$ at every step) are both
proved by induction on the FL step count.
\end{lemma}
\begin{proof}\rocqok
Long induction. The hard part is \texttt{fl\_divisibility\_L2}: the
$k$-th Newton power-sum identity
$p_k = \sum_{i=1}^{k-1}(-1)^{i-1} e_i p_{k-i} + (-1)^{k-1} k e_k$
is the algebraic content. We use the explicit recurrence form rather
than the global identity, which lets the induction track each FL step
in lockstep.
\end{proof}

\begin{theorem}[FL correctness over $\Q$]\label{thm:char_poly_int_correct}
\rocq{PrimeGapS1.CharPoly.char_poly_int_correct}
\rocqok
\uses{lem:fl_loop_rat_is_charpoly}
For a square integer matrix $M$ of dimension $n$,
\[
\texttt{pol\_to\_polyrat}(\texttt{char\_poly\_int}(M))
   \;=\; \texttt{char\_poly}(\texttt{mat\_int\_to\_rat}(M, 1, n))
   \quad\text{in } \Q[X].
\]
\end{theorem}
\begin{proof}\rocqok
Combine Lemma~\ref{lem:fl_loop_rat_is_charpoly} with the bridging
definitions \texttt{Z\_to\_int}, \texttt{mat\_int\_to\_rat},
\texttt{pol\_to\_polyrat}: each FL step on the rational side equals
the standard MathComp \texttt{char\_poly} computation by
\texttt{Newton-identities-charpoly}. The induction is over the matrix
dimension; at the inductive step we use that the leading coefficient
of \texttt{char\_poly} is~$1$ and that the next-to-leading coefficient
is $-\trace M$, both already in MathComp.
\end{proof}

\section{The \texorpdfstring{$c \cdot M$}{c.M} scaling formula}

The cleared-denominator polynomial $\texttt{charpoly\_int}$ shipped in
\texttt{Witness.v} satisfies
$\texttt{charpoly\_int} = \Dq \cdot \charpolysym_{\Arat}$ \emph{up to} a
power of $\DA$ that comes from $\Arat = \Aint / \DA$. The bridge is the
following one-lemma file.

\begin{lemma}[Char-poly scaling]\label{lem:char_poly_scale}
\rocq{PrimeGapS1.CharPolyScale.char_poly_scale}
\rocqok
For $F$ a field, $M : 'M[F]_n$, $c \in F$ with $c \ne 0$, and
$k \le n$:
\[
\bigl(\charpolysym(c \cdot M)\bigr)_k
   \;=\; c^{n - k} \cdot \bigl(\charpolysym(M)\bigr)_k.
\]
\end{lemma}
\begin{proof}\rocqok
Factor $c$ out of $xI - cM = c \cdot ((x/c) I - M)$. Since
$\charpolysym(M)(x) = \det(xI - M)$ and $\det$ is multilinear in rows,
\[
\det(xI - cM) = c^n \det\!\bigl((x/c) I - M\bigr) = c^n \cdot \charpolysym(M)(x/c).
\]
Reading off the coefficient of $x^k$ on the right via
\texttt{coef\_comp\_scaleX} gives
$c^n \cdot c^{-k} \cdot (\charpolysym(M))_k = c^{n - k} (\charpolysym(M))_k$,
which is what we wanted. The proof uses MathComp's
\texttt{det\_map\_mx} plus a direct coefficient computation.
\end{proof}

\chapter{CRT cross-validation}\label{ch:crt}

The most computationally heavy step in the entire proof is showing that
$\texttt{char\_poly\_int}(\Aint) = \texttt{charpoly\_of\_A\_int}$, the
shipped char-poly of the $42 \times 42$ integer matrix $\Aint$. Both
sides are lists of $43$ integers with coefficients up to ${\sim}20{,}000$
bits. Direct \vmcompute{} on the LHS would take hours and produce
unwieldy proof terms. Instead we use a Chinese Remainder Theorem (CRT)
lift: verify the equality modulo $710$ Uint63 primes, then close
$|a - b| = 0$ from $(\prod p_i) \mid (a - b)$ via a verified bound. The
same pattern handles the matrix identity $\Mtwo \cdot \Mone \cdot \Aint
= \DMone \cdot \DA \cdot \Mtwo$ that links the cleared denominators.
This chapter follows files \texttt{ModularArith.v}, \texttt{CRTBridge.v},
the \texttt{CharPolyAgree/} directory (\texttt{Def.v} + six parallel
\texttt{Chunk\_0..5.v} + the assembly \texttt{CharPolyAgree.v}),
\texttt{CRTCheck.v}, \texttt{Fermat.v}, \texttt{PrimeCheck.v} and
\texttt{CRTLift.v}.

\section{Modular arithmetic in Uint63}

\begin{definition}[Modular FL char-poly]\label{def:char_poly_mod}
\rocq{PrimeGapS1.ModularArith.char_poly_mod}
\rocqok
\uses{def:fl_loop}
\texttt{char\_poly\_mod}~$p\,M$ runs the same FL recurrence as
$\texttt{char\_poly\_int}$ but modulo a single $\Uintsf$ prime $p$,
using only native 63-bit operations: \texttt{addmod63},
\texttt{mulmod63}, \texttt{negmod63}, plus modular inverse
\texttt{inv\_mod63}~$p\,a := a^{p - 2} \bmod p$ (Fermat's little
theorem). For $p < 2^{31}$, all intermediate products $a \cdot b$ fit
in $2^{62} < 2^{63}$, so no overflow occurs and \vmcompute{}
processes a single $42 \times 42$ FL chain in ${\sim}0.5$~s.
\end{definition}

\begin{remark}[Why \texttt{ModularArith.v} exists as one file]
\label{rem:modular_arith_factor}
The original development duplicated all modular operations across
\texttt{CRTBridge.v} and \texttt{CharPolyAgree.v}. When
\texttt{CRTLift.v} tried to rewrite a fact mentioning
\texttt{char\_poly\_mod}~$p\,\Aint$ from one file by an equation
mentioning the same expression but defined in the other file, the
kernel saw two distinct head constants and tried to delta-unfold both,
which is exactly the FL recurrence on a concrete $42 \times 42$
matrix. Empirically that hung the kernel for $> 25$ minutes per
rewrite. Factoring every modular operation into one file pins each
constant to a single kernel name, so unification of the same head
symbol is by reflexivity.
\end{remark}

\section{Per-prime correctness of modular FL}

\begin{theorem}[Soundness of \texttt{char\_poly\_mod}]\label{thm:char_poly_mod_sound}
\rocq{PrimeGapS1.CRTBridge.char_poly_mod_sound}
\rocqok
\uses{def:char_poly_mod, def:fl_loop}
For any $\Uintsf$ prime $p$ with $1 < p < 2^{31}$ and $n + 1 < p$
(where $n$ is the matrix dimension), and any \texttt{fl\_all\_divisible}
matrix~$M$,
\[
\texttt{List.map}\,(\texttt{Z\_to\_mod63}\,p)\,(\texttt{char\_poly\_int}\,M)
   \;=\; \texttt{char\_poly\_mod}\,p\,M.
\]
The hypothesis $n + 1 < p$ ensures every FL division by $k \in \{1, \ldots, n\}$
is well-defined in $\Fp$.
\end{theorem}
\begin{proof}\rocqok
Thread soundness lemmas for each modular operation through the FL
recurrence. The base step is $\texttt{Z\_to\_mod63\_spec}$, which
proves
$\texttt{to\_Z}(\texttt{Z\_to\_mod63}\,p\,z) = z \bmod \texttt{to\_Z}\,p$
under \texttt{valid\_prime}~$p$ (precisely the
$2^{31}$-bound that guarantees no $\Uintsf$ multiplication overflow).
The inductive step combines $\texttt{addmod63}$/$\texttt{mulmod63}$
soundness with the FL invariants from
Lemma~\ref{lem:fl_loop_rat_is_charpoly}, plus Fermat's little theorem
(below) to discharge the modular division.
\end{proof}

\begin{lemma}[Fermat's little theorem, modular inverse form]
\label{lem:fermat_mod}
\rocq{PrimeGapS1.Fermat.fermat_mod, PrimeGapS1.Fermat.fermat_Z}
\rocqok
For prime $p$ and $1 \le k < p$,
$k \cdot k^{p - 2} \equiv 1 \pmod{p}$.
The $\Z$-level twin \texttt{fermat\_Z} bridges to
\texttt{Znumtheory.prime}; together with
\texttt{Zprime\_to\_ssrprime} this provides the interface that the
710-prime CRT lift consumes.
\end{lemma}
\begin{proof}\rocqok
Via MathComp's \texttt{expf\_card} on the multiplicative group $\Fp^\ast$.
\end{proof}

\rocq{PrimeGapS1.CharPolyAgree.crt_primes_local,

\rocqok
A concatenated list \texttt{crt\_primes\_all} of $710$ distinct

\begin{lemma}[All 710 candidates are prime]\label{lem:crt_primes_all_prime}
\rocq{PrimeGapS1.CRTLift.crt_primes_710_all_prime,
      PrimeGapS1.CRTLift.crt_primes_710_NoDup}
\rocqok
\uses{def:crt_primes}
Every entry of \texttt{crt\_primes\_all} is a prime number, and the

\section{The CRT lift}

\begin{lemma}[Small multiple is zero]\label{lem:small_multiple_zero}
\rocq{PrimeGapS1.CRTCheck.small_multiple_zero}
\rocqok
If $P \mid c$, $0 < P$, and $2|c| < P$ (all in $\Z$), then $c = 0$.
\end{lemma}
\begin{proof}\rocqok
A divisor of $c$ has absolute value $\le |c|$ unless $c = 0$. If
$c \ne 0$ then $P \le |c| < P/2$, contradiction. \texttt{lia} closes
the integer arithmetic.
\end{proof}

\begin{lemma}[Distinct primes divide the product]\label{lem:all_primes_divide}
\rocq{PrimeGapS1.CRTCheck.all_primes_divide_product}
\rocqok
For a no-duplicate list \texttt{ps} of primes, if $p \mid c$ for every
$p \in \texttt{ps}$, then $\prod\texttt{ps} \mid c$.
\end{lemma}
\begin{proof}\rocqok
Induction on \texttt{ps}. The inductive step uses the elementary fact
that distinct primes are coprime, hence $p \mid c$ and $q \mid c$
imply $pq \mid c$ when $\gcd(p, q) = 1$. The \texttt{NoDup} hypothesis
is precisely what guarantees pairwise coprimality at each step.
\end{proof}

\begin{theorem}[CRT lift: shipped char-poly is correct]
\label{thm:fl_eq_flint}
\rocq{PrimeGapS1.CRTLift.fl_eq_flint}
\rocqok
\uses{def:char_poly_mod, thm:char_poly_mod_sound, def:crt_primes,
      lem:crt_primes_all_prime, lem:small_multiple_zero,
      lem:all_primes_divide}
$\texttt{char\_poly\_int}(\Aint) = \texttt{charpoly\_of\_A\_int}$
in $\texttt{list }\Z$.
\end{theorem}
\begin{proof}\rocqok
Both sides are lists of $43$ integers. We prove equality
coefficient-by-coefficient by reducing
$a_n - b_n = 0$ to Lemma~\ref{lem:small_multiple_zero}.

For each prime $p$ in \texttt{crt\_primes\_all},
Theorem~\ref{thm:char_poly_mod_sound} gives
$\texttt{List.map}\,(\texttt{Z\_to\_mod63}\,p)\,
   (\texttt{char\_poly\_int}\,\Aint)
   = \texttt{char\_poly\_mod}\,p\,\Aint$.
On the other hand, a single-prime \vmcompute{} check inside
\texttt{check\_charpoly\_710} (a Boolean check that all $710$ per-prime
agreements hold simultaneously, closed by \vmcompute{} +
\texttt{reflexivity}) gives
$\texttt{char\_poly\_mod}\,p\,\Aint
   = \texttt{List.map}\,(\texttt{bigZ\_to\_mod63}\,p)\,
       \texttt{charpoly\_of\_A\_int\_bigZ}$.
Concatenating these two facts, every coefficient of the
left-hand side and the right-hand side has the same residue modulo
each $p$; hence $p \mid (a_n - b_n)$. By
Lemma~\ref{lem:all_primes_divide}, $\prod p \mid (a_n - b_n)$.

The bound $|a_n|, |b_n| \le B$ is established by
\texttt{charpoly\_coeff\_bound} (which tracks the FL recurrence's
own coefficient growth via \texttt{fl\_bound\_aux}) for $a_n$ and
\texttt{max\_abs\_coeff\_bound} for $b_n$. The combined bound
$2B + 2|c'| < \prod p_i$ is verified by \vmcompute{} in
\texttt{fl\_crt\_bound}, the single dominant step of the entire
build (${\sim}18$~min wall-clock comparing a ${\sim}10000$-digit
arithmetic chain). Lemma~\ref{lem:small_multiple_zero} then closes
$a_n = b_n$, and \texttt{nth\_ext} aggregates the per-coefficient
equalities into list equality.
\end{proof}

\begin{theorem}[Matrix identity in $\Z$]\label{thm:matrix_identity_Z}
\rocq{PrimeGapS1.CRTLift.matrix_identity_Z}
\rocqok
\uses{def:char_poly_mod, def:crt_primes, lem:small_multiple_zero,
      lem:all_primes_divide}
\[
\DMtwo \cdot (\Mone^{\mathrm{int}} \cdot \Aint)
   \;=\; (\DMone \cdot \DA) \cdot \Mtwo^{\mathrm{int}}
   \quad\text{in } \texttt{mat}.
\]
\end{theorem}
\begin{proof}\rocqok
Identical CRT pattern to Theorem~\ref{thm:fl_eq_flint}, but the
coefficient bound is much simpler because the entries are not the
output of a long recurrence: each entry of either side is bounded by
$|D| \cdot \texttt{max\_abs\_entry}$ products of three matrices of
fixed dimension. The same $710$-prime product covers it with room
to spare.
\end{proof}

\begin{remark}[\texttt{Strategy opaque} as a kernel-engineering tool]
\label{rem:strategy_opaque}
A naive proof of the per-prime fact
$\texttt{char\_poly\_mod}\,p\,\Aint
   = \texttt{List.map}\,(\texttt{bigZ\_to\_mod63}\,p)\,
       \texttt{charpoly\_of\_A\_int\_bigZ}$
hangs the \Qed{} for $> 25$~minutes: the kernel's ordinary reducer
tries to iota-reduce \texttt{list\_eqb63} and forces WHNF on
$\texttt{char\_poly\_mod}\,p\,\Aint$, which triggers the full FL
recurrence on a concrete $42 \times 42$ matrix. The fix is to wrap
the \Qed{} in
\texttt{Strategy opaque [list\_eqb63 char\_poly\_mod $\Aint$ \ldots]},
forcing the kernel to compare head symbols structurally, not unfold
the values. This is mathematically a no-op but operationally
indispensable.
\end{remark}

\chapter{The Sturm-chain root extraction}\label{ch:sturm}

This chapter constructs an explicit real algebraic root of
$\texttt{charpoly\_int}$ above $4/105$. The strategy is the
\emph{intermediate value theorem} on a real-closed field: we show
$P(4/105) < 0$ and $P(\text{cauchy\_bound}) > 0$ from the
shipped sign vectors, and let MathComp's \texttt{poly\_ivtoo} produce
a root in the interval. The material lives in \texttt{SignChain.v},
\texttt{CRTSigns.v}, \texttt{Smoke.v}, \texttt{Bridge.v} and
\texttt{CertL1.v}.

\section{The shipped Sturm chain}

The 43-polynomial Sturm chain that the FLINT layer ships in

headline IVT proof reads only entry $0$ of the chain, anchored to

\begin{lemma}[Chain anchors to \texttt{charpoly\_int}]

\rocq{PrimeGapS1.Smoke.chain_0_matches_charpoly}
\rocqok
\uses{def:sturm_chain}
$\texttt{sturm\_chain}\,[0] = \texttt{charpoly\_int}$,

characteristic polynomial.
\end{lemma}

\end{proof}

For every $i$, the $i$-th entry of \texttt{signs\_at\_x0} equals

\section{IVT: a root exists above \texorpdfstring{$4/105$}{4/105}}

\begin{theorem}[L1: IVT root]\label{thm:maynard_L1}
\rocq{PrimeGapS1.CertL1.maynard_L1_concrete}
\rocqok
\uses{def:sturm_chain, lem:signs_at_x0, lem:signs_at_inf,
      lem:chain_0_matches}
There exists $\lambda \in \realalg$ such that
\[
\texttt{root}\,(\texttt{pol\_to\_polyralg}\,\texttt{charpoly\_int})\,\lambda
\quad\text{and}\quad
\texttt{ratr}(4/105) < \lambda.
\]
\end{theorem}
\begin{proof}\rocqok
The proof rests on MathComp's intermediate value theorem
\texttt{poly\_ivtoo} on a real-closed field. We need two sign facts at
the endpoints:

\textbf{Lower endpoint.} The first entry of \texttt{signs\_at\_x0} is
$-1$ (verified by \vmcompute{} on the $\Z$-list). By
Lemma~\ref{lem:signs_at_x0}, this is $\sgn(P_0(4/105))$ where
$P_0 = \texttt{charpoly\_int}$, hence $P_0(4/105) < 0$.

\textbf{Upper endpoint.} Take $cb := \texttt{cauchy\_bound}(P_0)$,
MathComp's explicit rational upper bound for all real roots of $P_0$
derived from its coefficients. Two facts: $P_0 \ne 0$ (it has degree
$42$) and the first entry of \texttt{signs\_at\_inf} is $+1$ (by
Lemma~\ref{lem:signs_at_inf}, $\sgn(\lcoeff(P_0)) = +1$). Since $cb$
is a strict upper bound for all roots, $\sgn(P_0(cb)) = \sgn(\lcoeff(P_0)) = +1$,
hence $P_0(cb) > 0$.

\texttt{poly\_ivtoo} then yields $x \in [4/105, cb]$ with $P_0(x) = 0$.
Since $P_0(4/105) < 0$ but $P_0(x) = 0$, $x \ne 4/105$, hence
$4/105 < x$. Setting $\lambda := x$ closes the existential.
\end{proof}

\begin{remark}
This is \emph{IVT}, not the full Sturm theorem: we only need the
existence of some root above the threshold, not the exact count. The
Sturm count $V(4/105) - V(+\infty) = 1$ confirms uniqueness but is
mathematically superfluous for $L_4$. The bridge from
\texttt{list Z}-level signs to MathComp's
$\texttt{sgp\_pinfty}$ / polynomial-value signs is supplied by
\texttt{Bridge.v} (\texttt{pol\_to\_polyralg}, the
$\texttt{sgn\_matches}$-style lemmas, \texttt{cauchy\_bound}
boilerplate).
\end{remark}

\section{\texorpdfstring{$\Arat$ is a well-defined $\Q$-matrix}{A\_rat is a well-defined Q-matrix}}

\begin{lemma}[$\Mone$ is invertible over $\Q$]\label{lem:M1_unit}
\rocq{PrimeGapS1.CertL2.M1_1_unit}

\section{Linking the cleared polynomial to \texorpdfstring{$\charpolysym_{\Arat}$}{chi\_A\_rat}}

\begin{lemma}[Structural matrix scaling]
\label{lem:mat_A_scale_eq_Arat}
\rocq{PrimeGapS1.CertL2.mat_A_scale_eq_Arat,
      PrimeGapS1.CertL2.abstract_mat_scale}
\rocqok
\uses{def:A_rat, lem:M1_unit, thm:matrix_identity_Z}
$\texttt{mat\_int\_to\_rat}(\Aint, 1, 42) = (\DA)_{\Q} \cdot \Arat$
in $'M[\Q]_{42}$.
\end{lemma}
\begin{proof}\rocqok
The structural identity holds at any field $F$ and any dimension $n$:
given $\Mone, \Mtwo, A : 'M[F]_n$ with $c_2 \cdot (\Mone \cdot A) = (c_1 c_A) \cdot \Mtwo$
and $\Mone$ invertible, one solves
$A = c_A \cdot (c_1^{-1} \Mone)^{-1} \cdot (c_2^{-1} \Mtwo)$.
This is proved once at generic $F$, $n$ in
\texttt{abstract\_mat\_scale}, then specialised to $(F, n) = (\Q, 42)$.

The reason for the abstraction is purely operational: at concrete
$'M[\Q]_{42}$, MathComp's HB canonical-structure resolver traverses
an instance graph quadratic in the goal term, taking ${\sim}40$--$90$~min
per rewrite. At generic $F$ and $n$, the instance graph is
small ($F$ and $n$ are opaque), so elaboration is instantaneous.
The hypothesis $c_2 \cdot (\Mone \cdot A) = (c_1 c_A) \cdot \Mtwo$ is
discharged at the call site by Theorem~\ref{thm:matrix_identity_Z}
(the $\Z$-level matrix identity, lifted to $\Q$ via
\texttt{mat\_int\_to\_rat}).
\end{proof}

\begin{theorem}[L2: cleared polynomial is $\Dq \cdot \charpolysym_{\Arat}$]
\label{thm:charpoly_int_Dq}
\rocq{PrimeGapS1.CertL2.charpoly_int_Dq_scaled,
      PrimeGapS1.CertL2.char_poly_scale_rat42}
\rocqok
\uses{thm:char_poly_int_correct, thm:fl_eq_flint,
      lem:mat_A_scale_eq_Arat, lem:char_poly_scale}
\[
\texttt{pol\_to\_polyrat}(\texttt{charpoly\_int})
   \;=\; (\Dq)_{\Q} \cdot \charpolysym(\Arat)
   \quad\text{in } \Q[X].
\]
\end{theorem}
\begin{proof}\rocqok
We chain four facts:
\begin{enumerate}
\item Theorem~\ref{thm:fl_eq_flint}:
  $\texttt{char\_poly\_int}(\Aint) = \texttt{charpoly\_of\_A\_int}$
  (CRT lift).
\item Theorem~\ref{thm:char_poly_int_correct}:
  $\texttt{pol\_to\_polyrat}(\texttt{char\_poly\_int}(\Aint))
   = \charpolysym(\texttt{mat\_int\_to\_rat}(\Aint, 1, 42))$.
\item Lemma~\ref{lem:mat_A_scale_eq_Arat}:
  $\texttt{mat\_int\_to\_rat}(\Aint, 1, 42) = (\DA)_{\Q} \cdot \Arat$.
\item Lemma~\ref{lem:char_poly_scale}:
  $(\charpolysym(c \cdot M))_k = c^{n - k} \cdot (\charpolysym(M))_k$.
\end{enumerate}
A direct combination at the polynomial level gives
\[
\texttt{pol\_to\_polyrat}(\texttt{charpoly\_int})
   = (\Dq)_{\Q} \cdot \charpolysym(\Arat),
\]
where $\Dq$ absorbs the per-coefficient scaling factors $\DA^{42 - k}$
through the per-coefficient identity \texttt{scaling\_Z\_from\_check}
(verified via the same CRT route in \texttt{CharPolyAgree.v}).

The proof is closed by term-level rewriting (avoiding the slow
\texttt{rewrite} that would force HB elaboration on $'M[\Q]_{42}$):
the rewritten hypothesis is built explicitly via
\texttt{eq\_trans}~$\circ$~\texttt{f\_equal} chains, and
Lemma~\ref{lem:char_poly_scale} is invoked through the
pre-specialised \texttt{char\_poly\_scale\_rat42}. Both techniques
keep the concrete $'M[\Q]_{42}$ out of the tactic-level goal, where
HB resolution would stall for hours.
\end{proof}

\section{The headline theorem}

\begin{lemma}[L3: char-poly root $\Rightarrow$ eigenvalue]
\label{lem:eigenvalue_of_root}
\rocq{PrimeGapS1.Cert.eigenvalue_of_root_realalg}
\rocqok
For any $\lambda \in \realalg$, if
$\texttt{root}\,(\texttt{map\_poly}\,\texttt{ratr}\,\charpolysym(\Arat))\,\lambda$
then $\texttt{eigenvalue}\,(\texttt{map\_mx}\,\texttt{ratr}\,\Arat)\,\lambda$.
\end{lemma}
\begin{proof}\rocqok
A two-line script. \texttt{map\_char\_poly} (a MathComp lemma) says
$\charpolysym(\texttt{map\_mx}\,f\,M) = \texttt{map\_poly}\,f\,(\charpolysym\,M)$
for any ring morphism $f$; \texttt{eigenvalue\_root\_char} (also
MathComp) says an eigenvalue is exactly a root of the char poly.
Composing the two gives the implication.
\end{proof}

\begin{lemma}[L2': root transfer]\label{lem:charpoly_root_transfer}
\rocq{PrimeGapS1.Cert.charpoly_root_transfer}
\rocqok
\uses{thm:charpoly_int_Dq}
For $\lambda \in \realalg$, if
$\texttt{root}\,(\texttt{pol\_to\_polyralg}\,\texttt{charpoly\_int})\,\lambda$
then
$\texttt{root}\,(\texttt{map\_poly}\,\texttt{ratr}\,\charpolysym(\Arat))\,\lambda$.
\end{lemma}
\begin{proof}\rocqok
By Theorem~\ref{thm:charpoly_int_Dq},
$\texttt{pol\_to\_polyrat}\,\texttt{charpoly\_int}
   = \Dq \cdot \charpolysym(\Arat)$.
Mapping through $\texttt{ratr}$ pulls the scalar out
($\texttt{map\_polyZ}$), and \texttt{rootZ} says a nonzero scalar does
not kill roots. Nonzero-ness of $\Dq$ is checked inline by
\vmcompute{} on the shipped constant.
\end{proof}

\begin{lemma}[L4: rescale $4 < 105 \lambda$]\label{lem:bridge_L4}
\rocq{PrimeGapS1.Cert.maynard_bridge_L4}
\rocqok
For $\lambda \in \realalg$ with $\lambda > 0$,
$4 < 105 \lambda \iff 4/105 < \lambda$.
\end{lemma}
\begin{proof}\rocqok
A schoolbook \texttt{ltr\_pdivrMr} rewrite. Stated separately for
narrative clarity; the headline Theorem~\ref{thm:headline} states the
$4/105 < \lambda$ form directly.
\end{proof}

\begin{theorem}[Spectral headline: an eigenvalue exceeds $4/105$]
\label{thm:headline}
\rocq{PrimeGapS1.Cert.maynard_eigenvalue_S1,
      PrimeGapS1.Cert.sturm_count_correct}
\rocqok
\uses{thm:maynard_L1, lem:charpoly_root_transfer,
      lem:eigenvalue_of_root}
There exists $\lambda \in \realalg$ such that
\[
\texttt{eigenvalue}\,(\texttt{map\_mx}\,\texttt{ratr}\,\Arat)\,\lambda
\quad\text{and}\quad
\texttt{ratr}(4/105) < \lambda.
\]
\end{theorem}
\begin{proof}\rocqok
A four-line tactic. From Theorem~\ref{thm:maynard_L1}
(\texttt{sturm\_count\_correct} is a thin wrapper) we obtain
$\lambda \in \realalg$ that is a root of
$\texttt{pol\_to\_polyralg}\,\texttt{charpoly\_int}$ and satisfies
$\texttt{ratr}(4/105) < \lambda$. Lemma~\ref{lem:charpoly_root_transfer}
transfers root-ness from $\texttt{charpoly\_int}$ to $\charpolysym(\Arat)$.
Lemma~\ref{lem:eigenvalue_of_root} converts a char-poly root to an
eigenvalue of $\texttt{map\_mx}\,\texttt{ratr}\,\Arat$. The strict
inequality is preserved verbatim from $L_1$. Combining gives the
existential.
\end{proof}

\begin{theorem}[End-to-end certification]\label{thm:M105}
\rocq{PrimeGapS1.Cert.maynard_M105_certified}
\rocqok
\uses{thm:M1_match, thm:M2_match, thm:headline}
The conjunction
\[
\texttt{all\_match\_M1Z} = \texttt{true}
\;\wedge\;
\texttt{all\_match\_M2Z} = \texttt{true}
\;\wedge\;
\bigl(\exists \lambda \in \realalg.\;
   \texttt{eigenvalue}(\texttt{map\_mx}\,\texttt{ratr}\,\Arat)\,\lambda
   \;\wedge\;
   \texttt{ratr}(4/105) < \lambda\bigr).
\]
\end{theorem}
\begin{proof}\rocqok
Three-way \texttt{split} over the bool facts
\texttt{all\_match\_M1Z\_true}, \texttt{all\_match\_M2Z\_true} and
the spectral existence Theorem~\ref{thm:headline}. The combined
\texttt{Print Assumptions} reports only the standard Rocq
\texttt{PrimInt63} kernel primitives (used by \vmcompute{} on the
$\Uintsf$ CRT computations). No project-specific axioms.
\end{proof}

\paragraph{Reading.}
A reviewer who trusts \emph{only Rocq's kernel} obtains, from
Theorem~\ref{thm:M105}, a certified existence of a $\realalg$
eigenvalue of a specific kernel-verified $42 \times 42$ $\Q$-matrix
above $4/105$. A reviewer who additionally accepts Maynard's Lemma~8.3
(an Annals-published Rayleigh-quotient identity, paper-side) obtains
$\Mk[105] > 4$ end-to-end.

\chapter{Trust base}\label{ch:trust}

\section{What is a Qed and what is paper-side}

\begin{itemize}
\item All claims of the form
  $a = b$ in $\Z$ or $\Q$, $\det \ne 0$, root of a polynomial,
  eigenvalue of a $\Q$-matrix, or sign of an algebraic-real polynomial
  evaluation, that appear in the dependency tree of
  Theorem~\ref{thm:M105} are \Qed{} in the source tree.
  Zero \texttt{Axiom}, zero \texttt{Admitted}.
\item The matrix entries themselves (Lemma~8.2 of Maynard) are
  \Qed{} via Theorems~\ref{thm:M1_match} and~\ref{thm:M2_match}: the
  $1764 + 1764$ kernel-checked cross-multiplications close the trust
  loop on the input data.
\item Maynard's Lemma~8.3 ($\Mk = k \cdot \lammax$) is the
  generalised Rayleigh-quotient identity that bridges
  $\lammax > 4/k$ to $\Mk > 4$. This is the analytic step proved in
  the Annals paper (refereed there). It is \emph{not} formalised
  in this repository, by design of the project's scope.
\item The reality of the spectrum of $\Mone^{-1}\Mtwo$ (a consequence
  of $\Mone$ being symmetric positive-definite and $\Mtwo$ symmetric)
  is also paper-side. The Rocq layer extracts a real algebraic
  eigenvalue via IVT (which is automatically real), so it does not
  need to formalise the structural reality claim.
\item The analytic Beta-integral derivation that produces Maynard's
  closed-form $M_1$ and $M_2$ entries (Lemma~8.1 / 8.2 in the paper) is
  taken as a \emph{definition} in Rocq. The kernel certifies that the
  shipped integer matrices match those closed forms, not that the
  closed forms themselves are correct integrations.
\end{itemize}

\section{Reported assumptions}

\texttt{Print Assumptions maynard\_M105\_certified} (or
\texttt{maynard\_eigenvalue\_S1}) reports only the Rocq~9.0/9.1 kernel
primitives for native 63-bit integers and \texttt{BigN}: things like
$\texttt{Uint63.add}$, $\texttt{Uint63.mul}$, $\texttt{Uint63.to\_Z}$,
$\texttt{BigN.succ\_spec}$. These are not project-specific axioms;
they are Rocq's declaration that the OCaml-implemented native-integer
kernel primitives implement their stated specifications. Any proof
that uses \vmcompute{} on $\Uintsf$ exposes the same set.

The project does not invoke \texttt{native\_compute} anywhere — a
\texttt{grep -n native\_compute theories/S1} returns zero hits in real
proof code. This keeps the trust-base enumeration minimal at the cost
of a slightly longer build (the dominant \texttt{fl\_crt\_bound}
\vmcompute{} runs in ${\sim}18$~min instead of ${\sim}2$~min with
\texttt{native\_compute}).

\section{The FLINT layer is outside the trust base}

The FLINT pipeline (Python + \texttt{python-flint}) is the candidate
generator for the certificate data and an independent cross-check, but
the Rocq proof never invokes Python and never loads the JSON
certificates directly. The autogenerated \texttt{Witness.v} and
\texttt{WitnessChain.v} are ordinary Rocq files; if the FLINT layer
shipped wrong data, one of the following \vmcompute{}-based checks
would fail:
\begin{itemize}
\item Theorems~\ref{thm:M1_match} and~\ref{thm:M2_match}: the
  $42 \times 42$ matrix entries match Maynard's closed form
  (Lemma~8.2).
\item \texttt{check\_charpoly\_710}: the shipped char-poly is the
  char poly of $\Aint$ modulo each of $710$ primes.
\item \texttt{matrix\_identity\_710}: the shipped denominator-clearing
  matrix identity holds modulo each of $710$ primes.
\item Lemmas~\ref{lem:signs_at_x0} and~\ref{lem:signs_at_inf}: the
  shipped sign vectors agree with direct \BigZ\ evaluation on the
  shipped Sturm chain.
\item Lemma~\ref{lem:chain_0_matches}: $\texttt{chain}[0] =
  \texttt{charpoly\_int}$ (the only chain entry the IVT proof reads).
\end{itemize}
Each is a \Qed{} closed by pure kernel arithmetic.

\chapter{Numerical highlights}\label{ch:numeric}

\paragraph{Matrix dimension.} $42 \times 42$.
Maynard's $\Mk[105]$ construction uses $42$ basis polynomials
$\{F_{b,c} : b + 2c \le 11\}$.

\paragraph{Char-poly degree.} $42$ (so $43$ coefficients), with
coefficients up to ${\sim}20{,}000$ bits.

\paragraph{CRT primes.} $710$ \Uintsf\ primes in $[2^{30}, 2^{31})$,
with product exceeding $2^{21300}$ — well above the verified bound
$2B < 2^{6500}$ needed for the CRT lift.

\paragraph{Denominators.} $\DMone$ has ${\sim}221$ decimal digits,
$\DMtwo$ ${\sim}227$, $\DA$ ${\sim}139$, $\Dq$ ${\sim}333$. All four
are strictly positive (the others follow from being products of
factorials; positivity of $\Dq$ is checked inline by \vmcompute{} in
\texttt{charpoly\_root\_transfer}).

\paragraph{Cauchy bound.} The IVT step uses MathComp's
\texttt{cauchy\_bound} — an explicit rational upper bound on all real
roots derived from the polynomial coefficients. No custom bound is
introduced.

\paragraph{Build profile.}
\texttt{make -j} on a multi-core box: ${\sim}37$~min wall-clock
(${\sim}80$~min sequential). The dominant single-\Qed{} costs are:
\begin{itemize}
\item \texttt{fl\_crt\_bound}: ${\sim}18$~min — the
  $2B + 2|c'| < \prod p_i$ comparison with ${\sim}10{,}000$-digit
  arithmetic.
\item Theorem~\ref{thm:M2_match}
  (\texttt{all\_match\_M2Z\_true}): ${\sim}35$~min — $1764$ entries,
  each up to $36$ rational summands with intermediate denominators
  reaching $\sim 10^{7000}$ digits.
\item Lemma~\ref{lem:crt_primes_all_prime}
  (\texttt{crt\_primes\_710\_all\_prime}): ${\sim}7$~min — $710$
  pure-$\Z$ trial-division primality checks.
\item \texttt{check\_charpoly\_710}, \texttt{matrix\_identity\_710}:
  ${\sim}5$--$10$~min each — $710 \times 42$-step modular FL.
\item Theorem~\ref{thm:M1_match}: ${\sim}90$~s.
\item \texttt{Witness.v}/\texttt{WitnessChain.v} parsing: ${\sim}1$~min
  — \BigZ\ literal parser on ${\sim}20$~MB of shipped integers.
\end{itemize}
With \texttt{make -j} the M2 \vmcompute{} runs concurrently with the
CRT primality and 710-FL phases, so it does not strictly add
$35$ minutes to the wall clock — but it dominates the tail.

\paragraph{Project size.} $26$ \texttt{.v} files, ${\sim}19{,}200$
lines total: ${\sim}7{,}500$ autogenerated certificate data
(\texttt{Witness.v} + \texttt{WitnessChain.v}) and ${\sim}11{,}700$
hand-written proof script.

\chapter{Map of key lemmas and files}\label{ch:map}

The dependency graph has two independent backbones merging at
\texttt{Cert.v}:

\paragraph{Sturm-chain backbone.}
\texttt{Witness} $\to$ \texttt{WitnessChain} $\to$
\texttt{IntPoly} / \texttt{SignChain} $\to$
\texttt{CRTSigns} $\to$ \texttt{Bridge} $\to$ \texttt{CertL1}.

\paragraph{CharPoly backbone.}
\texttt{Witness} $\to$ \texttt{IntMat} / \texttt{IntPoly} $\to$
\texttt{CharPoly} $\to$ \texttt{ModularArith} $\to$
\texttt{CharPolyAgree} + \texttt{CRTBridge} $\to$ \texttt{CRTLift}
$\to$ \texttt{CharPolyScale} $\to$ \texttt{CertL2}.
Headline output: Theorem~\ref{thm:charpoly_int_Dq}.

\paragraph{Maynard-spec backbone.}
\texttt{Witness} $\to$ \texttt{MaynardFactQ} $\to$
\texttt{MaynardBasis} + \texttt{MaynardSpec} $\to$
\texttt{MaynardVerify}. Headline output:
Theorems~\ref{thm:M1_match} and~\ref{thm:M2_match}.

The three backbones merge at \texttt{Cert.v}, producing
Theorems~\ref{thm:headline} and~\ref{thm:M105}. A single
\texttt{Require Import PrimeGapS1.Cert.} loads the full proof.
