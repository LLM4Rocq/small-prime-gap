% Math macros and theorem environments shared by web.tex and print.tex.
%
% The dep-graph collects environments definition, lemma, proposition,
% theorem and corollary by default. We add `proposition` for L0/L4-style
% summary statements.

\newtheorem{theorem}{Theorem}
\newtheorem{proposition}[theorem]{Proposition}
\newtheorem{lemma}[theorem]{Lemma}
\newtheorem{corollary}[theorem]{Corollary}

\theoremstyle{definition}
\newtheorem{definition}[theorem]{Definition}

\theoremstyle{remark}
\newtheorem*{remark}{Remark}
\newtheorem*{convention}{Convention}

% ---------------------------------------------------------------------
% Project-specific math macros.
%
% Usage convention: every macro below is called from inside math mode
% (i.e., between $...$ or inside an equation environment) — VERIFIED
% by an audit of content.tex. So they are defined as plain math-mode
% commands. Wrapping them in \ensuremath{} is tempting but breaks
% interactions with unicode-math when used as superscripts or
% subscripts (e.g. M_k^\Q triggers a "Missing { inserted" because the
% \ensuremath expansion confuses unicode-math's __um_group_begin).
%
% The two exceptions below (\Uintsf, \BigZ) ARE called from text mode
% in several places ("the \Uintsf kernel", "$43$ \BigZ\ coefficients")
% and are sans-serif, never used in super/subscripts. They use
% \ensuremath to be safe in both contexts.
% ---------------------------------------------------------------------
